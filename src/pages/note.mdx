  {(posts) => (
    <Grid
      sx={{
        gridTemplateColumns: ["1fr", "1fr", "1fr 1fr 1fr"],
      }}
    >
      {posts
        .filter((edge) => !edge.node.frontmatter.isPrivate)
        .reduce((posts, post) => {
          return post.node.frontmatter.pinned
            ? [post, ...posts]
            : [...posts, post];
        }, [])
        .map((edge, index) => {
          const {
            featuredImageUrl,
            frontmatter: { title, tags, date, dateModified, pinned },
            excerpt,
            fields: { slug },
          } = edge.node;
          return (
            <ProjectCard
              key={index}
              title={title}
              featuredImageUrl={featuredImageUrl}
              tags={tags}
              date={date}
              dateModified={dateModified}
              excerpt={excerpt}
              slug={slug}
              pinned={pinned}
            />
          );
        })}
    </Grid>
